Qemu shell commands
    # shows register dump
    - info registers
    
    # Dumps number of bytes/instructions in specified format, at specified address
    - xp /<count><format><size> <address> 
        <format> 
        - "d", signed decimal
        - "u", unsigned decimal
        - "o", octal
        - "x", hex
        <size>
        - "b", 8 bits 
        - "h", 16 bits
        - "w", 32 bits
        - "g", 64 bits
